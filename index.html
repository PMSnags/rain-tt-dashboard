Apps Script Setup — Enable Write-Back to Google Sheets

Follow these steps once to enable saving TT Status and Remarks directly to your Google Sheet.

Open your Google Sheet → click Extensions → Apps Script
Delete all existing code, then paste the script below and click Save (Ctrl+S)
Click Deploy → New Deployment → Type: Web App → Execute as: Me → Who has access: Anyone → click Deploy
Copy the Web App URL, paste it below and click Connect

Copy and paste this into Apps Script (Extensions → Apps Script):

// ══════════════════════════════════════════════════════════════════ // DIGITUS Rain TT Dashboard — Apps Script v3 // IMPORTANT: Replace ALL existing code with this. Then: // Deploy > New Deployment > Web App // Execute as: Me | Who has access: Anyone // Copy the NEW Web App URL and paste it in the dashboard. // ══════════════════════════════════════════════════════════════════ var SHEET_NAME = ' Rain TTs Tracker'; // ← must match your tab name exactly (note leading space) var LOG_NAME = 'Update Log'; // ← auto-created if missing // All dashboard calls come through doGet (JSONP works with GET, not POST) function doGet(e) { var p = e.parameter; var cb = p.callback || 'callback'; try { if (p.action === 'save') return wrap(cb, handleSave(p)); if (p.action === 'getLog') return wrap(cb, handleGetLog()); if (p.action === 'ping') return wrap(cb, {pong: true, status: 'online'}); return wrap(cb, {status: 'online', version: '3.0'}); } catch(err) { return wrap(cb, {result: 'error', msg: err.toString()}); } } function handleSave(p) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return {result: 'error', msg: 'Sheet not found: [' + SHEET_NAME + ']'}; var rows = sheet.getDataRange().getValues(); var headers = rows[0].map(function(h){ return String(h).trim(); }); var ttNumCol = headers.indexOf('TT Number'); var statusCol = headers.indexOf('TT Status'); var remCol = headers.indexOf('Additional remarks'); var siteCol = headers.indexOf('Site ID'); if (ttNumCol < 0) return {result: 'error', msg: '"TT Number" column not found'}; for (var i = 1; i < rows.length; i++) { if (String(rows[i][ttNumCol]).trim() === String(p.ttNumber).trim()) { var oldStatus = String(rows[i][statusCol] || ''); var oldRemarks = String(rows[i][remCol] || ''); if (statusCol >= 0 && p.status !== undefined) sheet.getRange(i+1, statusCol+1).setValue(p.status); if (remCol >= 0 && p.remarks !== undefined) sheet.getRange(i+1, remCol+1).setValue(p.remarks); writeLog(ss, { timestamp : new Date(), ttNumber : p.ttNumber, siteId : rows[i][siteCol] || '', updatedBy : p.updatedBy || 'Dashboard User', oldStatus : oldStatus, newStatus : p.status !== undefined ? p.status : oldStatus, oldRemarks: oldRemarks, newRemarks: p.remarks !== undefined ? p.remarks : oldRemarks }); return {result: 'ok'}; } } return {result: 'not_found'}; } function handleGetLog() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var log = ss.getSheetByName(LOG_NAME); if (!log) return {result: 'ok', log: []}; var rows = log.getDataRange().getValues(); var headers = rows[0]; var data = rows.slice(1).reverse().map(function(r) { var obj = {}; headers.forEach(function(h, i) { obj[String(h)] = String(r[i] || ''); }); return obj; }); return {result: 'ok', log: data}; } function writeLog(ss, entry) { var log = ss.getSheetByName(LOG_NAME); if (!log) { log = ss.insertSheet(LOG_NAME); var hdr = log.getRange(1, 1, 1, 8); log.appendRow(['Timestamp','TT Number','Site ID','Updated By', 'Old Status','New Status','Old Remarks','New Remarks']); hdr.setFontWeight('bold').setBackground('#1a0305').setFontColor('#ffffff'); log.setFrozenRows(1); log.setColumnWidth(1, 160); } log.appendRow([ entry.timestamp, entry.ttNumber, entry.siteId, entry.updatedBy, entry.oldStatus, entry.newStatus, entry.oldRemarks, entry.newRemarks ]); } function wrap(cb, obj) { return ContentService .createTextOutput(cb + '(' + JSON.stringify(obj) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } // doPost kept as fallback (not used by dashboard) function doPost(e) { return doGet(e); } // ══════════════════════════════════════════════════════════════════
